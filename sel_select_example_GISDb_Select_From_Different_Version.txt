// THIS IS AN EXAMPLE HOW TO SELECT IN MS SQL SERVER A GIS TABLE WHICH IS VERSIONED.
//SET VERSION TO A DIFFERENT VERSION
EXEC sde.set_current_version 'MyDifferntVersionThatIsNotDefault'
//DO STANDARD SQL SELECT STATEMENT
SELECT MAX([SEGMENTID])
  FROM [MyGISDb].[GISDBOwner].[MyTableStreetCenterline_VW];
GO
//SET VERSION BACK TO DEFAULT FOR OTHER TASKS.
EXEC sde.set_default
